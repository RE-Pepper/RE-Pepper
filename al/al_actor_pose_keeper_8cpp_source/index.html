void calcSideDir( sead::Vector3f* out, const LiveActor* actor )
{
        sead::Matrix34f baseMtx;
        actor->getActorPoseKeeper()->calcBaseMtx( &baseMtx );
        out->x = baseMtx.m[ 0 ][ 0 ];
        out->y = baseMtx.m[ 1 ][ 0 ];
        out->z = baseMtx.m[ 2 ][ 0 ];
}

void calcUpDir( sead::Vector3f* out, const LiveActor* actor )
{
        sead::Matrix34f baseMtx;
        actor->getActorPoseKeeper()->calcBaseMtx( &baseMtx );
        out->x = baseMtx.m[ 0 ][ 1 ];

        out->z = baseMtx.m[ 2 ][ 1 ];
}
